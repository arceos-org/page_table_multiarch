3use core::arch::asm;

9/// Metadata of ARMv7-A page tables.
10pub struct A32PagingMetaData;
11
12impl PagingMetaData for A32PagingMetaData {
13    const LEVELS: usize = 2; // ARMv7-A uses 2-level page tables
14    const PA_MAX_BITS: usize = 32;
15    const VA_MAX_BITS: usize = 32;
16    type VirtAddr = memory_addr::VirtAddr;
17
18    fn vaddr_is_valid(_vaddr: usize) -> bool {
19        // All 32-bit addresses are valid
20        //     vaddr < 0xFFFF_FFFF
21        true
22    }
23
24    #[inline]
25    fn flush_tlb(vaddr: Option<memory_addr::VirtAddr>) {
26        unsafe {
27            if let Some(vaddr) = vaddr {
28                // Invalidate unified TLB entry by MVA
29                asm!(
30                    "mcr p15, 0, {0}, c8, c7, 1", // TLBIMVA
31                    in(reg) vaddr.as_usize(),
32                );
33            } else {
34                // Invalidate entire unified TLB
35                let zero: usize = 0;
36                asm!(
37                    "mcr p15, 0, {0}, c8, c7, 0", // TLBIALL
38                    in(reg) zero,
39                );
40            }
41            // Data Synchronization Barrier
42            asm!("dsb");
43            // Instruction Synchronization Barrier
44            asm!("isb");
45        }
46    }
47}
48
49/// ARMv7-A Short-descriptor translation table.
50pub type A32PageTable<H> = PageTable32<A32PagingMetaData, A32PTE, H>;
51/// ARMv7-A translation table cursor.
52pub type A32PageCursor<'a, H> = PageTable32Cursor<'a, A32PagingMetaData, A32PTE, H>;
