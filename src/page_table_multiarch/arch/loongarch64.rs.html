7/// Metadata of LoongArch64 page tables.
8#[derive(Copy, Clone, Debug)]
9pub struct LA64MetaData;
10
11impl LA64MetaData {
12    /// PWCL(Page Walk Controller for Lower Half Address Space) CSR flags
13    ///
14    /// <https://loongson.github.io/LoongArch-Documentation/LoongArch-Vol1-EN.html#page-walk-controller-for-lower-half-address-space>
15    ///
16    /// | BitRange | Name      | Value |
17    /// | ----     | ----      | ----  |
18    /// | 4:0      | PTBase    |    12 |
19    /// | 9:5      | PTWidth   |     9 |
20    /// | 14:10    | Dir1Base  |    21 |
21    /// | 19:15    | Dir1Width |     9 |
22    /// | 24:20    | Dir2Base  |    30 |
23    /// | 29:25    | Dir2Width |     9 |
24    /// | 31:30    | PTEWidth  |     0 |
25    pub const PWCL_VALUE: u32 = 12 | (9 << 5) | (21 << 10) | (9 << 15) | (30 << 20) | (9 << 25);
26
27    /// PWCH(Page Walk Controller for Higher Half Address Space) CSR flags
28    ///
29    /// <https://loongson.github.io/LoongArch-Documentation/LoongArch-Vol1-EN.html#page-walk-controller-for-higher-half-address-space>
30    ///
31    /// | BitRange | Name                            | Value |
32    /// | ----     | ----                            | ----  |
33    /// | 5:0      | Dir3Base                        |    39 |
34    /// | 11:6     | Dir3Width                       |     9 |
35    /// | 17:12    | Dir4Base                        |     0 |
36    /// | 23:18    | Dir4Width                       |     0 |
37    /// | 24       | 0                               |     0 |
38    /// | 24       | HPTW_En(CPUCFG.2.HPTW(bit24)=1) |     0 |
39    /// | 31:25    | 0                               |     0 |
40    pub const PWCH_VALUE: u32 = 39 | (9 << 6);
41}
42
43impl PagingMetaData for LA64MetaData {
44    const LEVELS: usize = 4;
45    const PA_MAX_BITS: usize = 48;
46    const VA_MAX_BITS: usize = 48;
47    type VirtAddr = memory_addr::VirtAddr;
48
49    #[inline]
50    fn flush_tlb(vaddr: Option<memory_addr::VirtAddr>) {
51        unsafe {
52            if let Some(vaddr) = vaddr {
53                // <https://loongson.github.io/LoongArch-Documentation/LoongArch-Vol1-EN.html#_dbar>
54                //
55                // Only after all previous load/store access operations are completely
56                // executed, the DBAR 0 instruction can be executed; and only after the
57                // execution of DBAR 0 is completed, all subsequent load/store access
58                // operations can be executed.
59                //
60                // <https://loongson.github.io/LoongArch-Documentation/LoongArch-Vol1-EN.html#_invtlb>
61                //
62                // formats: invtlb op, asid, addr
63                //
64                // op 0x5: Clear all page table entries with G=0 and ASID equal to the
65                // register specified ASID, and VA equal to the register specified VA.
66                //
67                // When the operation indicated by op does not require an ASID, the
68                // general register rj should be set to r0.
69                asm!("dbar 0; invtlb 0x05, $r0, {reg}", reg = in(reg) vaddr.as_usize());
70            } else {
71                // op 0x0: Clear all page table entries
72                asm!("dbar 0; invtlb 0x00, $r0, $r0");
73            }
74        }
75    }
76}
