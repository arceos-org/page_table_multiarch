1//! AArch64 specific page table structures.
2
3use core::arch::asm;
4use page_table_entry::aarch64::A64PTE;
5
6use crate::{PageTable64, PagingMetaData};
7
8/// Metadata of AArch64 page tables.
9pub struct A64PagingMetaData;
10
11impl PagingMetaData for A64PagingMetaData {
12    const LEVELS: usize = 4;
13    const PA_MAX_BITS: usize = 48;
14    const VA_MAX_BITS: usize = 48;
15    type VirtAddr = memory_addr::VirtAddr;
16
17    fn vaddr_is_valid(vaddr: usize) -> bool {
18        let top_bits = vaddr >> Self::VA_MAX_BITS;
19        top_bits == 0 || top_bits == 0xffff
20    }
21
22    #[inline]
23    fn flush_tlb(vaddr: Option<memory_addr::VirtAddr>) {
24        unsafe {
25            if let Some(vaddr) = vaddr {
26                // TLB Invalidate by VA, All ASID, EL1, Inner Shareable
27                const VA_MASK: usize = (1 << 44) - 1; // VA[55:12] => bits[43:0]
28                asm!("tlbi vaae1is, {}; dsb sy; isb", in(reg) ((vaddr.as_usize() >> 12) & VA_MASK))
29            } else {
30                // TLB Invalidate by VMID, All at stage 1, EL1
31                asm!("tlbi vmalle1; dsb sy; isb")
32            }
33        }
34    }
35}
36
37/// AArch64 VMSAv8-64 translation table.
38pub type A64PageTable<H> = PageTable64<A64PagingMetaData, A64PTE, H>;
