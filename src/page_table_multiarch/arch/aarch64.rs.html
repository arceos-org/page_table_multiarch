10/// Metadata of AArch64 page tables.
11pub struct A64PagingMetaData;
12
13impl PagingMetaData for A64PagingMetaData {
14    const LEVELS: usize = 4;
15    const PA_MAX_BITS: usize = 48;
16    const VA_MAX_BITS: usize = 48;
17
18    type VirtAddr = VirtAddr;
19
20    fn vaddr_is_valid(vaddr: usize) -> bool {
21        let top_bits = vaddr >> Self::VA_MAX_BITS;
22        top_bits == 0 || top_bits == 0xffff
23    }
24
25    #[inline]
26    fn flush_tlb(vaddr: Option<VirtAddr>) {
27        unsafe {
28            if let Some(vaddr) = vaddr {
29                // TLB Invalidate by VA, All ASID, EL1, Inner Shareable
30                const VA_MASK: usize = (1 << 44) - 1; // VA[55:12] => bits[43:0]
31                asm!("tlbi vaae1is, {}; dsb sy; isb", in(reg) ((vaddr.as_usize() >> 12) & VA_MASK))
32            } else {
33                // TLB Invalidate by VMID, All at stage 1, EL1
34                asm!("tlbi vmalle1; dsb sy; isb")
35            }
36        }
37    }
38}

40/// AArch64 VMSAv8-64 translation table.
41pub type A64PageTable<H> = PageTable64<A64PagingMetaData, A64PTE, H>;
42/// AArch64 VMSAv8-64 translation table cursor.
43pub type A64PageTableCursor<'a, H> = PageTable64Cursor<'a, A64PagingMetaData, A64PTE, H>;
