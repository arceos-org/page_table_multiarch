11#[cfg(target_arch = "arm")]
12const ENTRY_COUNT: usize = 4096; // ARMv7-A L1 has 4096 entries
13#[cfg(not(target_arch = "arm"))]
14const ENTRY_COUNT: usize = 512; // 512 entries per table
15
16/// Extract the L1 (first-level) page table index from a virtual address.
17///
18/// For ARMv7-A:
19/// - L1 uses bits[31:20] of the virtual address (12 bits = 4096 entries)
20/// - Each L1 entry covers 1MB of virtual address space
21const fn p1_index(vaddr: usize) -> usize {
22    (vaddr >> 20) & 0xFFF // bits[31:20] for 1MB sections
23}
24
25/// Extract the L2 (second-level) page table index from a virtual address.
26///
27/// For ARMv7-A:
28/// - L2 uses bits[19:12] of the virtual address (8 bits = 256 entries)
29/// - Each L2 entry covers 4KB of virtual address space
30const fn p2_index(vaddr: usize) -> usize {
31    (vaddr >> 12) & 0xFF // bits[19:12] for 4KB pages
32}

34/// A generic page table struct for 32-bit ARM platform (ARMv7-A).
35///
36/// This implements a 2-level page table:
37/// - L1: 4096 entries, each covering 1MB (Section) or pointing to L2
38/// - L2: 256 entries, each covering 4KB (Small Page)
39///
40/// It tracks all L2 tables for proper deallocation.
41pub struct PageTable32<M: PagingMetaData, PTE: GenericPTE, H: PagingHandler> {
42    root_paddr: PhysAddr,
43    #[cfg(feature = "copy-from")]
44    borrowed_entries: [u64; ENTRY_COUNT / 64],
45    _phantom: PhantomData<(M, PTE, H)>,
46}

48impl<M: PagingMetaData, PTE: GenericPTE, H: PagingHandler> PageTable32<M, PTE, H> {
49    /// Creates a new page table instance or returns the error.
50    ///
51    /// It will allocate a new 16KB aligned page for the L1 page table.
52    pub fn try_new() -> PagingResult<Self> {
53        let (root_paddr, size_pages) = {
54            #[cfg(target_arch = "arm")]
55            {
56                // ARMv7-A L1 page table: 4096 entries * 4 bytes = 16KB
57                // Requires 16KB alignment for TTBR0
58                const L1_SIZE_PAGES: usize = 4; // 16KB = 4 * 4KB
59                const L1_ALIGN: usize = 16384; // 16KB alignment
60
61                let root_paddr =
62                    H::alloc_frames(L1_SIZE_PAGES, L1_ALIGN).ok_or(PagingError::NoMemory)?;
63
64                (root_paddr, L1_SIZE_PAGES)
65            }
66
67            #[cfg(not(target_arch = "arm"))]
68            {
69                // Other 32-bit architectures page table: 512 entries * 8 bytes = 4KB
70                const SIZE_PAGES: usize = 1; // 4KB = 1 * 4KB
71                let root_paddr = H::alloc_frame().ok_or(PagingError::NoMemory)?;
72                (root_paddr, SIZE_PAGES)
73            }
74        };
75
76        // Zero out the root page table
77        let virt = H::phys_to_virt(root_paddr);
78        unsafe {
79            core::ptr::write_bytes(virt.as_mut_ptr(), 0, size_pages * PAGE_SIZE_4K);
80        }
81
82        Ok(Self {
83            root_paddr,
84            #[cfg(feature = "copy-from")]
85            borrowed_entries: [0; ENTRY_COUNT / 64],
86            _phantom: PhantomData,
87        })
88    }
89
90    /// Returns the physical address of the root page table (L1).
91    pub const fn root_paddr(&self) -> PhysAddr {
92        self.root_paddr
93    }
94
95    /// Query the result of the mapping starts with `vaddr`.
96    ///
97    /// Returns the physical address of the target frame, the mapping flags, and
98    /// the page size.
99    ///
100    /// Returns [`Err(PagingError::NotMapped)`](PagingError::NotMapped) if the
101    /// mapping is not present.
102    pub fn query(&self, vaddr: M::VirtAddr) -> PagingResult<(PhysAddr, MappingFlags, PageSize)> {
103        let (entry, size) = self.get_entry(vaddr)?;
104        if entry.is_unused() {
105            return Err(PagingError::NotMapped);
106        }
107        let off = vaddr.into() & (size as usize - 1);
108        Ok((entry.paddr().add(off), entry.flags(), size))
109    }
110
111    /// Walk the page table recursively.
112    pub fn walk<F>(&self, limit: usize, pre_func: Option<&F>, post_func: Option<&F>)
113    where
114        F: Fn(usize, usize, M::VirtAddr, &PTE),
115    {
116        self.walk_recursive(
117            self.get_table(self.root_paddr),
118            0,
119            0.into(),
120            limit,
121            pre_func,
122            post_func,
123        )
124    }
125
126    /// Gets a cursor to modify the page table.
127    ///
128    /// The TLB will be flushed automatically when the cursor is dropped.
129    pub fn cursor(&mut self) -> PageTable32Cursor<'_, M, PTE, H> {
130        PageTable32Cursor::new(self)
131    }

133    // Private helpers
134    fn get_entry_mut(&mut self, vaddr: M::VirtAddr) -> PagingResult<(&mut PTE, PageSize)> {
135        let vaddr_usize = vaddr.into();
136        let p1 = p1_index(vaddr_usize);
137        let table = self.get_table_mut(self.root_paddr);
138        let entry = &mut table[p1];
139
140        if entry.is_unused() {
141            return Err(PagingError::NotMapped);
142        }
143
144        // Check if it's a 1MB Section
145        if entry.is_huge() {
146            return Ok((entry, PageSize::Size1M));
147        }
148
149        // It's a page table pointer, go to L2
150        let p2_table = self.get_table_mut(entry.paddr());
151        let p2 = p2_index(vaddr_usize);
152        Ok((&mut p2_table[p2], PageSize::Size4K))
153    }
154
155    fn get_entry(&self, vaddr: M::VirtAddr) -> PagingResult<(&PTE, PageSize)> {
156        let vaddr_usize = vaddr.into();
157        let p1 = p1_index(vaddr_usize);
158        let table = self.get_table(self.root_paddr);
159        let entry = &table[p1];
160
161        if entry.is_unused() {
162            return Err(PagingError::NotMapped);
163        }
164
165        if entry.is_huge() {
166            return Ok((entry, PageSize::Size1M));
167        }
168
169        let p2_table = self.get_table(entry.paddr());
170        let p2 = p2_index(vaddr_usize);
171        Ok((&p2_table[p2], PageSize::Size4K))
172    }
173
174    fn get_entry_mut_or_create(
175        &mut self,
176        vaddr: M::VirtAddr,
177        page_size: PageSize,
178    ) -> PagingResult<&mut PTE> {
179        let vaddr_usize = vaddr.into();
180        let p1 = p1_index(vaddr_usize);
181        let table = self.get_table_mut(self.root_paddr);
182
183        if page_size == PageSize::Size1M {
184            // Map as 1MB Section in L1
185            return Ok(&mut table[p1]);
186        }
187
188        // Need L2 page table for 4KB mapping
189        let entry = &mut table[p1];
190        if entry.is_unused() {
191            // Create new L2 page table (allocate 4KB, though only 1KB is used)
192            let paddr = H::alloc_frame().ok_or(PagingError::NoMemory)?;
193
194            // Zero out the L2 page table
195            let virt = H::phys_to_virt(paddr);
196            unsafe {
197                core::ptr::write_bytes(virt.as_mut_ptr(), 0, PAGE_SIZE_4K);
198            }
199
200            *entry = GenericPTE::new_table(paddr);
201        } else if entry.is_huge() {
202            // Already mapped as huge page
203            return Err(PagingError::AlreadyMapped);
204        }
205
206        let p2_table = self.get_table_mut(entry.paddr());
207        let p2 = p2_index(vaddr_usize);
208        Ok(&mut p2_table[p2])
209    }
210
211    fn get_table<'a>(&self, paddr: PhysAddr) -> &'a [PTE] {
212        let ptr = H::phys_to_virt(paddr).as_ptr() as *const PTE;
213        unsafe { core::slice::from_raw_parts(ptr, ENTRY_COUNT) }
214    }
215
216    fn get_table_mut<'a>(&self, paddr: PhysAddr) -> &'a mut [PTE] {
217        let ptr = H::phys_to_virt(paddr).as_mut_ptr() as *mut PTE;
218        unsafe { core::slice::from_raw_parts_mut(ptr, ENTRY_COUNT) }
219    }
220
221    fn walk_recursive<F>(
222        &self,
223        table: &[PTE],
224        level: usize,
225        start_vaddr: M::VirtAddr,
226        limit: usize,
227        pre_func: Option<&F>,
228        post_func: Option<&F>,
229    ) where
230        F: Fn(usize, usize, M::VirtAddr, &PTE),
231    {
232        let start_vaddr_usize: usize = start_vaddr.into();
233        let mut n = 0;
234        for (i, entry) in table.iter().enumerate() {
235            // L1 (level 0): each entry covers 1MB (shift 20)
236            // L2 (level 1): each entry covers 4KB (shift 12)
237            let shift = if level == 0 { 20 } else { 12 };
238            let vaddr_usize = start_vaddr_usize + (i << shift);
239            let vaddr = vaddr_usize.into();
240
241            if !entry.is_unused() {
242                if let Some(func) = pre_func {
243                    func(level, i, vaddr, entry);
244                }
245                if level == 0 && !entry.is_huge() {
246                    let next_table = self.get_table(entry.paddr());
247                    self.walk_recursive(next_table, level + 1, vaddr, limit, pre_func, post_func);
248                }
249                if let Some(func) = post_func {
250                    func(level, i, vaddr, entry);
251                }
252                n += 1;
253                if n >= limit {
254                    break;
255                }
256            }
257        }
258    }
259}

261impl<M: PagingMetaData, PTE: GenericPTE, H: PagingHandler> Drop for PageTable32<M, PTE, H> {
262    fn drop(&mut self) {
263        // Deallocate all L2 page tables (each is 4KB)
264        let table = self.get_table(self.root_paddr);
265        #[allow(unused_variables)]
266        for (i, entry) in table.iter().enumerate() {
267            #[cfg(feature = "copy-from")]
268            if (self.borrowed_entries[i / 64] & (1 << (i % 64))) != 0 {
269                continue;
270            }
271            if !entry.is_unused() && !entry.is_huge() {
272                // This is an L2 page table (4KB)
273                H::dealloc_frame(entry.paddr());
274            }
275        }
276        // Deallocate L1 page table (16KB = 4 pages)
277        H::dealloc_frames(self.root_paddr, 4);
278    }
279}

297impl<'a, M: PagingMetaData, PTE: GenericPTE, H: PagingHandler> PageTable32Cursor<'a, M, PTE, H> {
298    fn new(inner: &'a mut PageTable32<M, PTE, H>) -> Self {
299        Self {
300            inner,
301            flusher: TlbFlusher::None,
302        }
303    }
304
305    fn push(&mut self, vaddr: M::VirtAddr) {
306        match self.flusher {
307            TlbFlusher::None => {
308                let mut arr = ArrayVec::new();
309                arr.push(vaddr);
310                self.flusher = TlbFlusher::Array(arr);
311            }
312            TlbFlusher::Array(ref mut arr) => {
313                if arr.try_push(vaddr).is_err() {
314                    self.flusher = TlbFlusher::Full;
315                }
316            }
317            TlbFlusher::Full => {}
318        }
319    }
320
321    /// Maps a virtual page to a physical frame with the given `page_size`
322    /// and mapping `flags`.
323    pub fn map(
324        &mut self,
325        vaddr: M::VirtAddr,
326        target: PhysAddr,
327        page_size: PageSize,
328        flags: MappingFlags,
329    ) -> PagingResult {
330        let entry = self.inner.get_entry_mut_or_create(vaddr, page_size)?;
331        if !entry.is_unused() {
332            return Err(PagingError::AlreadyMapped);
333        }
334        *entry = GenericPTE::new_page(target.align_down(page_size), flags, page_size.is_huge());
335        self.push(vaddr);
336        Ok(())
337    }
338
339    /// Remaps the mapping starting at `vaddr`, updates both the physical
340    /// address and flags.
341    pub fn remap(
342        &mut self,
343        vaddr: M::VirtAddr,
344        paddr: PhysAddr,
345        flags: MappingFlags,
346    ) -> PagingResult<PageSize> {
347        let (entry, size) = self.inner.get_entry_mut(vaddr)?;
348        *entry = GenericPTE::new_page(paddr, flags, size.is_huge());
349        self.push(vaddr);
350        Ok(size)
351    }
352
353    /// Updates the flags of the mapping starting at `vaddr`.
354    pub fn protect(&mut self, vaddr: M::VirtAddr, flags: MappingFlags) -> PagingResult<PageSize> {
355        let (entry, size) = self.inner.get_entry_mut(vaddr)?;
356        if entry.is_unused() {
357            return Err(PagingError::NotMapped);
358        }
359        *entry = GenericPTE::new_page(entry.paddr(), flags, size.is_huge());
360        self.push(vaddr);
361        Ok(size)
362    }
363
364    /// Unmaps the mapping starting at `vaddr`.
365    pub fn unmap(
366        &mut self,
367        vaddr: M::VirtAddr,
368    ) -> PagingResult<(PhysAddr, MappingFlags, PageSize)> {
369        let (entry, size) = self.inner.get_entry_mut(vaddr)?;
370        if entry.is_unused() {
371            return Err(PagingError::NotMapped);
372        }
373        let paddr = entry.paddr();
374        let flags = entry.flags();
375        entry.clear();
376        self.push(vaddr);
377        Ok((paddr, flags, size))
378    }

380    /// Maps a contiguous virtual memory region to a contiguous physical memory
381    /// region with the given mapping `flags`.
382    pub fn map_region(
383        &mut self,
384        vaddr: M::VirtAddr,
385        get_paddr: impl Fn(M::VirtAddr) -> PhysAddr,
386        size: usize,
387        flags: MappingFlags,
388        allow_huge: bool,
389    ) -> PagingResult {
390        let mut vaddr_usize: usize = vaddr.into();
391        let mut size = size;
392        if !PageSize::Size4K.is_aligned(vaddr_usize) || !PageSize::Size4K.is_aligned(size) {
393            return Err(PagingError::NotAligned);
394        }
395        trace!(
396            "map_region({:#x}): [{:#x}, {:#x}) {:?}",
397            self.root_paddr(),
398            vaddr_usize,
399            vaddr_usize + size,
400            flags,
401        );
402        while size > 0 {
403            let vaddr = vaddr_usize.into();
404            let paddr = get_paddr(vaddr);
405            let page_size = if allow_huge
406                && PageSize::Size1M.is_aligned(vaddr_usize)
407                && paddr.is_aligned(PageSize::Size1M)
408                && size >= PageSize::Size1M as usize
409            {
410                PageSize::Size1M
411            } else {
412                PageSize::Size4K
413            };
414            self.map(vaddr, paddr, page_size, flags).inspect_err(|e| {
415                error!("failed to map page: {vaddr_usize:#x?}({page_size:?}) -> {paddr:#x?}, {e:?}")
416            })?;
417
418            vaddr_usize += page_size as usize;
419            size -= page_size as usize;
420        }
421        Ok(())
422    }
423
424    /// Unmaps a contiguous virtual memory region.
425    pub fn unmap_region(&mut self, vaddr: M::VirtAddr, size: usize) -> PagingResult {
426        let mut vaddr_usize: usize = vaddr.into();
427        let mut size = size;
428        trace!(
429            "unmap_region({:#x}) [{:#x}, {:#x})",
430            self.root_paddr(),
431            vaddr_usize,
432            vaddr_usize + size,
433        );
434        while size > 0 {
435            let vaddr = vaddr_usize.into();
436            let (_, _, page_size) = self
437                .unmap(vaddr)
438                .inspect_err(|e| error!("failed to unmap page: {vaddr_usize:#x?}, {e:?}"))?;
439
440            assert!(page_size.is_aligned(vaddr_usize));
441            assert!(page_size as usize <= size);
442            vaddr_usize += page_size as usize;
443            size -= page_size as usize;
444        }
445        Ok(())
446    }

448    /// Updates mapping flags of a contiguous virtual memory region.
449    pub fn protect_region(
450        &mut self,
451        vaddr: M::VirtAddr,
452        size: usize,
453        flags: MappingFlags,
454    ) -> PagingResult {
455        let mut vaddr_usize: usize = vaddr.into();
456        let mut size = size;
457        trace!(
458            "protect_region({:#x}) [{:#x}, {:#x}) {:?}",
459            self.root_paddr(),
460            vaddr_usize,
461            vaddr_usize + size,
462            flags,
463        );
464        while size > 0 {
465            let vaddr = vaddr_usize.into();
466            let page_size = match self.inner.get_entry_mut(vaddr) {
467                Ok((entry, page_size)) => {
468                    if !entry.is_unused() {
469                        entry.set_flags(flags, page_size.is_huge());
470                        self.push(vaddr);
471                    }
472                    // ignore if not present
473
474                    page_size
475                }
476                Err(PagingError::NotMapped) => PageSize::Size4K,
477                Err(e) => {
478                    error!("failed to protect page: {vaddr_usize:#x?}, {e:?}");
479                    return Err(e);
480                }
481            };
482
483            assert!(page_size.is_aligned(vaddr_usize));
484            assert!(page_size as usize <= size);
485            vaddr_usize += page_size as usize;
486            size -= page_size as usize;
487        }
488        Ok(())
489    }

491    /// Copy entries from another page table within the given virtual memory
492    /// range.
493    #[cfg(feature = "copy-from")]
494    pub fn copy_from(&mut self, other: &PageTable32<M, PTE, H>, start: M::VirtAddr, size: usize) {
495        if size == 0 {
496            return;
497        }
498        let src_table = self.inner.get_table(other.root_paddr);
499        let dst_table = self.inner.get_table_mut(self.inner.root_paddr);
500
501        let start_idx = p1_index(start.into());
502        let end_idx = p1_index(start.into() + size - 1) + 1;
503        assert!(start_idx < ENTRY_COUNT);
504        assert!(end_idx <= ENTRY_COUNT);
505
506        // Simple copy here, no smart flush or tracking borrowing for now in 32-bit
507        // The user just wants interface consistency.
508        for i in start_idx..end_idx {
509            let entry = &mut dst_table[i];
510            let is_borrowed = (self.inner.borrowed_entries[i / 64] & (1 << (i % 64))) != 0;
511            if !is_borrowed {
512                self.inner.borrowed_entries[i / 64] |= 1 << (i % 64);
513                if !entry.is_unused() && !entry.is_huge() {
514                    H::dealloc_frame(entry.paddr());
515                }
516            }
517            *entry = src_table[i];
518        }
519        self.flusher = TlbFlusher::Full;
520    }
521
522    /// Flushes the TLB according to the recorded flush requests.
523    pub fn flush(&mut self) {
524        #[cfg(not(docsrs))]
525        match &self.flusher {
526            TlbFlusher::None => {}
527            TlbFlusher::Array(addrs) => {
528                for vaddr in addrs.iter() {
529                    M::flush_tlb(Some(*vaddr));
530                }
531            }
532            TlbFlusher::Full => {
533                M::flush_tlb(None);
534            }
535        }
536        self.flusher = TlbFlusher::None;
537    }
538}
