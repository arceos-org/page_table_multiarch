3use core::fmt;
4use memory_addr::PhysAddr;

6pub use x86_64::structures::paging::page_table::PageTableFlags as PTF;
7
8use crate::{GenericPTE, MappingFlags};
9
10impl From<PTF> for MappingFlags {
11    fn from(f: PTF) -> Self {
12        if !f.contains(PTF::PRESENT) {
13            return Self::empty();
14        }
15        let mut ret = Self::READ;
16        if f.contains(PTF::WRITABLE) {
17            ret |= Self::WRITE;
18        }
19        if !f.contains(PTF::NO_EXECUTE) {
20            ret |= Self::EXECUTE;
21        }
22        if f.contains(PTF::USER_ACCESSIBLE) {
23            ret |= Self::USER;
24        }
25        if f.contains(PTF::NO_CACHE) {
26            ret |= Self::UNCACHED;
27        }
28        ret
29    }
30}
31
32impl From<MappingFlags> for PTF {
33    fn from(f: MappingFlags) -> Self {
34        if f.is_empty() {
35            return Self::empty();
36        }
37        let mut ret = Self::PRESENT;
38        if f.contains(MappingFlags::WRITE) {
39            ret |= Self::WRITABLE;
40        }
41        if !f.contains(MappingFlags::EXECUTE) {
42            ret |= Self::NO_EXECUTE;
43        }
44        if f.contains(MappingFlags::USER) {
45            ret |= Self::USER_ACCESSIBLE;
46        }
47        if f.contains(MappingFlags::DEVICE) || f.contains(MappingFlags::UNCACHED) {
48            ret |= Self::NO_CACHE | Self::WRITE_THROUGH;
49        }
50        ret
51    }
52}

54/// An x86_64 page table entry.
55#[derive(Clone, Copy)]
56#[repr(transparent)]
57pub struct X64PTE(u64);
58
59impl X64PTE {
60    const PHYS_ADDR_MASK: u64 = 0x000f_ffff_ffff_f000; // bits 12..52
61
62    /// Creates an empty descriptor with all bits set to zero.
63    pub const fn empty() -> Self {
64        Self(0)
65    }
66}
67
68impl GenericPTE for X64PTE {
69    fn new_page(paddr: PhysAddr, flags: MappingFlags, is_huge: bool) -> Self {
70        let mut flags = PTF::from(flags);
71        if is_huge {
72            flags |= PTF::HUGE_PAGE;
73        }
74        Self(flags.bits() | (paddr.as_usize() as u64 & Self::PHYS_ADDR_MASK))
75    }
76    fn new_table(paddr: PhysAddr) -> Self {
77        let flags = PTF::PRESENT | PTF::WRITABLE | PTF::USER_ACCESSIBLE;
78        Self(flags.bits() | (paddr.as_usize() as u64 & Self::PHYS_ADDR_MASK))
79    }
80    fn paddr(&self) -> PhysAddr {
81        PhysAddr::from((self.0 & Self::PHYS_ADDR_MASK) as usize)
82    }
83    fn flags(&self) -> MappingFlags {
84        PTF::from_bits_truncate(self.0).into()
85    }
86    fn set_paddr(&mut self, paddr: PhysAddr) {
87        self.0 = (self.0 & !Self::PHYS_ADDR_MASK) | (paddr.as_usize() as u64 & Self::PHYS_ADDR_MASK)
88    }
89    fn set_flags(&mut self, flags: MappingFlags, is_huge: bool) {
90        let mut flags = PTF::from(flags);
91        if is_huge {
92            flags |= PTF::HUGE_PAGE;
93        }
94        self.0 = (self.0 & Self::PHYS_ADDR_MASK) | flags.bits()
95    }
96
97    fn bits(self) -> usize {
98        self.0 as usize
99    }
100    fn is_unused(&self) -> bool {
101        self.0 == 0
102    }
103    fn is_present(&self) -> bool {
104        PTF::from_bits_truncate(self.0).contains(PTF::PRESENT)
105    }
106    fn is_huge(&self) -> bool {
107        PTF::from_bits_truncate(self.0).contains(PTF::HUGE_PAGE)
108    }
109    fn clear(&mut self) {
110        self.0 = 0
111    }
112}
113
114impl fmt::Debug for X64PTE {
115    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
116        let mut f = f.debug_struct("X64PTE");
117        f.field("raw", &self.0)
118            .field("paddr", &self.paddr())
119            .field("flags", &self.flags())
120            .finish()
121    }
